  const [nav, setNav] = useState(false);
  const [color, setColor] = useState('transparent');
  const [textColor, setTextColor] = useState('white');

    setNav(!nav);

            <Link href='/about'>About</Link>
          </li>

            <Link href='/notes'>Notes</Link>
          </li>

            <Link href='/contact'>Contact</Link>
